\subsection{Le code suivant prГ©sente le script Python utilisГ© pour nettoyer les textes extraits des fichiers EPUB aprГЁs leur conversion au format \textsl{.txt}}.

from pathlib import Path
import re


def nettoyage_structure_epub(texte):
    texte_propre = texte

    # Nettoyage initial des espaces insГ©cables
    texte_propre = re.sub(r'[\xa0\u202f]', ' ', texte_propre)

    # Suppression des Г©lГ©ments placГ©s entre crochets
    texte_propre = re.sub(r'\[.*?\]', '', texte_propre)

    # Conservation uniquement des caractГЁres utiles pour l'analyse
    texte_propre = re.sub(
        r'[^a-zA-Z0-9Г ГўГӨГ©ГЁГӘГ«Г®ГҜГҙГ¶Г№Г»ГјГ§ГҖГӮГ„ГүГҲГҠГӢГҺГҸГ”Г–ГҷГӣГңГҮЕ“Е’'
        r'\s.,;:!?\-()\[\]]',
        '',
        texte_propre
    )

    # Suppression des chiffres collГ©s aux mots
    texte_propre = re.sub(
        r'([a-zA-ZГ ГўГӨГ©ГЁГӘГ«Г®ГҜГҙГ¶Г№Г»ГјГ§ГҖГӮГ„ГүГҲГҠГӢГҺГҸГ”Г–ГҷГӣГңГҮЕ“Е’])\d+',
        r'\1',
        texte_propre
    )

    # Nettoyage des tirets et des guillemets
    texte_propre = re.sub(r'-{2,}', '-', texte_propre)
    texte_propre = re.sub(r'["В«В»вҖңвҖқ]', '', texte_propre)

    # Suppression des numГ©ros de page isolГ©s
    texte_propre = re.sub(r'(?m)^\s*\d+\s*$', '', texte_propre)

    # Suppression des titres de chapitre en chiffres romains
    texte_propre = re.sub(r"(?im)^\s*[IVXLCDM]+\s*$", "", texte_propre)

    # Suppression de certains Г©lГ©ments propres aux fichiers EPUB
    texte_propre = re.sub(r"(?im)^\s*Liste des titres\s*$", "", texte_propre)
    texte_propre = re.sub(r"(?im)^\s*Table des matiГЁres du titre\s*$", "", texte_propre)

    # Suppression des lignes entiГЁrement Г©crites en majuscules
    texte_propre = re.sub(
        r"(?m)^\s*[A-ZГҖГӮГ„ГүГҲГҠГӢГҺГҸГ”Г–ГҷГӣГңГҮ'вҖҷ\s]+\s*$",
        "", texte_propre
    )

    # Transformation du texte en un seul bloc continu
    texte_propre = re.sub(r'\s+', ' ', texte_propre)

    # Segmentation du texte en phrases
    pattern_segmentation = r"(?<!\bM)([.?!])\s+(?=[A-ZГҖГӮГ„ГүГҲГҠГӢГҺГҸГ”Г–ГҷГӣГңГҮ\-])"
    texte_propre = re.sub(pattern_segmentation, r"\1\n", texte_propre)

    # Remise Г  la ligne des dialogues
    texte_propre = re.sub(r'\s+(?=-)', r'\n', texte_propre)

    return texte_propre.strip()


# DГ©finition des dossiers source et destination
dossier_source = Path("nom_dossier")
dossier_destination = Path("Text_nettoyГ©/nom_dossier")

# CrГ©ation du dossier de destination s'il n'existe pas
dossier_destination.mkdir(parents=True, exist_ok=True)

# Compteur permettant de suivre le nombre de fichiers traitГ©s
fichiers_traites = 0

print(f"DГ©but du traitement des fichiers dans : {dossier_source}...\n")

# Boucle sur tous les fichiers .txt du dossier source
for fichier_source in dossier_source.glob("*.txt"):

    print(f"-> Nettoyage de : {fichier_source.name}")

    try:
        # Lecture du fichier brut
        contenu_brut = fichier_source.read_text(encoding="utf-8")

        # Nettoyage du texte
        contenu_propre = nettoyage_structure_epub(contenu_brut)

        # CrГ©ation du nom du fichier nettoyГ©
        nom_fichier_propre = f"{fichier_source.stem}_clean.txt"
        fichier_destination = dossier_destination / nom_fichier_propre

        # Sauvegarde du fichier nettoyГ©
        fichier_destination.write_text(contenu_propre, encoding="utf-8")

        fichiers_traites += 1

    except Exception as e:
        print(f"   [ERREUR] Impossible de traiter {fichier_source.name} : {e}")

print(
    f"\nOpГ©ration terminГ©e ! {fichiers_traites} fichiers ont Г©tГ© nettoyГ©s "
    f"et placГ©s dans le dossier '{dossier_destination}'."
)
